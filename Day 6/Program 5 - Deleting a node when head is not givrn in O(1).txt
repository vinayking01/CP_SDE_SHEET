/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    void deleteNode(ListNode* node) {
        ListNode* temp =node;
        ListNode* prev;
        while(temp->next!=NULL)
        {   prev = temp;
            temp->val = temp->next->val;
            temp = temp ->next;
        }
        delete temp;
        prev->next = NULL;
    }
};
